cmake_minimum_required(VERSION 2.8)
project(client_multibroadcast)
set( CMAKE_CXX_FLAGS "-std=c++11" )

IF (CMAKE_SYSTEM_NAME MATCHES "Linux")
    add_executable(${PROJECT_NAME} "../../src/src_client/client_multibroadcast_linux.cpp")
    target_link_libraries( client_multibroadcast )
ELSEIF (CMAKE_SYSTEM_NAME MATCHES "Windows")
    add_executable(${PROJECT_NAME} "../../src/src_client/client_multibroadcast_win.cpp")
    target_link_libraries(client_multibroadcast ws2_32)
ELSE ()
    MESSAGE(STATUS "platform error!")
ENDIF (CMAKE_SYSTEM_NAME MATCHES "Linux")
